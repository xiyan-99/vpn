function parseArgs(qs) {
  const out = {};
  if (!qs || typeof qs !== "string") return out;
  const s = qs.trim().replace(/^\?/, "");
  if (!s) return out;

  for (const part of s.split("&")) {
    if (!part) continue;
    const idx = part.indexOf("=");
    const k = decodeURIComponent((idx >= 0 ? part.slice(0, idx) : part).trim());

}
